async function startDownload() { const url = document.getElementById('url-input').value.trim(); if(!url) return; const dlId = Date.now(); tab('dl-manager', document.querySelectorAll('.nav-link')[1]); document.getElementById('active-list').innerHTML = `
প্রসেসিং হচ্ছে... (একটু সময় দিন)
`; try { const response = await fetch(`/download?url=${encodeURIComponent(url)}`); const result = await response.json(); // এপিআইভেদে ডাটার ফরম্যাট ভিন্ন হতে পারে, তাই সব চেক করা হচ্ছে const finalData = result.data || result; const videoUrl = finalData.url || (finalData.downloads && finalData.downloads[0].url); if (videoUrl) { document.getElementById(`bar-${dlId}`).style.width = "100%"; document.getElementById(`st-${dlId}`).innerText = "ডাউনলোড শুরু!"; const a = document.createElement('a'); a.href = videoUrl; a.download = "KR_Video_" + dlId + ".mp4"; a.click(); savedVids.unshift({ id: dlId, name: finalData.title || "Video_" + dlId, time: new Date().toLocaleTimeString(), url: videoUrl }); localStorage.setItem('kr_vids', JSON.stringify(savedVids)); renderVids(); } else { throw new Error(); } } catch (e) { document.getElementById(`st-${dlId}`).innerHTML = "দুঃখিত, এই ভিডিওটির লিঙ্ক পাওয়া যায়নি!"; } }
